% Custom Chapter, Section, and Subsection Formatting for the Table of Contents
\newcommand{\charterchapter}[1]{
	\chapter{#1}
}

\newcommand{\chartersection}[1]{
	\section*{#1}\addcontentsline{toc}{section}{#1}
}

\newcommand{\chartersubsection}[1]{
	\subsection*{#1}\addcontentsline{toc}{subsection}{#1}
}

\newcommand{\sectionitem}[1]{
	\addcontentsline{toc}{section}{#1}
	\item \textbf{#1}
}

\newcommand{\subsectionitem}[1]{
	\addcontentsline{toc}{subsection}{#1}
	\item \textbf{#1}
}

\newglossarystyle{charter}{%
	\setglossarystyle{list}%
	\renewcommand*{\glossentry}[2]{%
		\item[\glsentryitem{##1}%
		\glstarget{##1}{\glossentryname{##1}:}]
		\glossentrydesc{##1}\glspostdescription\space ##2}
}

% Ethereum Mainnet Addresses
\newcommand{\ethaddress}[1]{
	\href{https://etherscan.io/address/#1}{\texttt{#1}}
}
